<RICHTUNG>  ::= "LEFT" | "RIGHT" | "UP" | "DOWN"
<BUCHSTABE> ::= "A" | "B" | ... | "Z"

<ZIFFER>    ::= "0" | "1" | ... | "9"
<ZAHL>	    ::= <ZIFFER> | <ZIFFER> <ZAHL>




<OBSTACLE>	::= "OBSTACLE" | <BUCHSTABE>
<KLAMMER>	::= "{" <PROGRAM> "}"




<BEFEHL>	::= "MOVE" <RICHTUNG> | "COLLECT"

<WENN>		::= "IF" <RICHTUNG> "IS-A" <OBSTACLE> <KLAMMER>
<UNTIL>		::= "UNTIL" <RICHTUNG> "IS-A" <OBSTACLE> <KLAMMER>
<SCHLEIFE>	::= "REPEAT" <ZAHL> <KLAMMER>




<STATEMENT>	::= <BEFEHL> | <SCHLEIFE> | <UNTIL> | <WENN>

<PROGRAM>	::= <STATEMENT> | <STATEMENT> <PROGRAM>
