---

"SELECT"
<VARIABLE>  ::= "a" | "b" | … | "z"
"FROM" 
<TABLE>     ::= "users" | "data" 

[ "WHERE" ]
<BEDINGUNG> ::= ">" | "=" 
<ZIFFER>    ::= "0" | "1" | … | "9"

---

<WERT> ::= <VARIABLE> | <VARIABLE> "," <WERT> 

<WHERE> ::= <VARIABLE> <BEDINGUNG> <ZIFFER> 

<ABFRAGE> ::= "SELECT" <WERT> "FROM" <TABLE> [ "WHERE" <WHERE> ]
